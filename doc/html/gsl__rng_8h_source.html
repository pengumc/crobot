00191 gsl_rng_uniform_int (const gsl_rng * r, unsigned long int n)
00192 {
00193   unsigned long int offset = r->type->min;
00194   unsigned long int range = r->type->max - offset;
00195   unsigned long int scale;
00196   unsigned long int k;
00197 
00198   if (n > range || n == 0) 
00199     {
00200       GSL_ERROR_VAL ("invalid n, either 0 or exceeds maximum value of generator",
00201                      GSL_EINVAL, 0) ;
00202     }
00203 
00204   scale = range / n;
00205 
00206   do
00207     {
00208       k = (((r->type->get) (r->state)) - offset) / scale;
00209     }
00210   while (k >= n);
