00042 /* Description of a system of ODEs.
00043  *
00044  * y' = f(t,y) = dydt(t, y)
00045  *
00046  * The system is specified by giving the right-hand-side
00047  * of the equation and possibly a jacobian function.
00048  *
00049  * Some methods require the jacobian function, which calculates
00050  * the matrix dfdy and the vector dfdt. The matrix dfdy conforms
00051  * to the GSL standard, being a continuous range of floating point
00052  * values, in row-order.
00053  *
00054  * As with GSL function objects, user-supplied parameter
00055  * data is also present. 
00056  */
00057 
00058 typedef struct  
00059 {
00060   int (* function) (double t, const double y[], double dydt[], void * params);
00061   int (* jacobian) (double t, const double y[], double * dfdy, double dfdt[], void * params);
00062   size_t dimension;
00063   void * params;
00064 }
00065 gsl_odeiv_system;
00066 
00067 #define GSL_ODEIV_FN_EVAL(S,t,y,f)  (*((S)->function))(t,y,f,(S)->params)
00068 #define GSL_ODEIV_JA_EVAL(S,t,y,dfdy,dfdt)  (*((S)->jacobian))(t,y,dfdy,dfdt,(S)->params)

00176 /* Available control object constructors.
00177  *
00178  * The standard control object is a four parameter heuristic
00179  * defined as follows:
00180  *    D0 = eps_abs + eps_rel * (a_y |y| + a_dydt h |y'|)
00181  *    D1 = |yerr|
00182  *    q  = consistency order of method (q=4 for 4(5) embedded RK)
00183  *    S  = safety factor (0.9 say)
00184  *
00185  *                      /  (D0/D1)^(1/(q+1))  D0 >= D1
00186  *    h_NEW = S h_OLD * |
00187  *                      \  (D0/D1)^(1/q)      D0 < D1
00188  *
00189  * This encompasses all the standard error scaling methods.
00190  *
00191  * The y method is the standard method with a_y=1, a_dydt=0.
00192  * The yp method is the standard method with a_y=0, a_dydt=1.
00193  */
00194 
00195 gsl_odeiv_control * gsl_odeiv_control_standard_new(double eps_abs, double eps_rel, double a_y, double a_dydt);
00196 gsl_odeiv_control * gsl_odeiv_control_y_new(double eps_abs, double eps_rel);
00197 gsl_odeiv_control * gsl_odeiv_control_yp_new(double eps_abs, double eps_rel);
