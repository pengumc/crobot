00159 extern inline size_t
00160 gsl_interp_bsearch(const double x_array[], double x,
00161                    size_t index_lo, size_t index_hi)
00162 {
00163   size_t ilo = index_lo;
00164   size_t ihi = index_hi;
00165   while(ihi > ilo + 1) {
00166     size_t i = (ihi + ilo)/2;
00167     if(x_array[i] > x)
00168       ihi = i;
00169     else
00170       ilo = i;
00171   }
00172   
00173   return ilo;
00174 }
00175 #endif

00177 #ifdef HAVE_INLINE
00178 extern inline size_t
00179 gsl_interp_accel_find(gsl_interp_accel * a, const double xa[], size_t len, double x)
00180 {
00181   size_t x_index = a->cache;
00182  
00183   if(x < xa[x_index]) {
00184     a->miss_count++;
00185     a->cache = gsl_interp_bsearch(xa, x, 0, x_index);
00186   }
00187   else if(x > xa[x_index + 1]) {
00188     a->miss_count++;
00189     a->cache = gsl_interp_bsearch(xa, x, x_index, len-1);
00190   }
00191   else {
00192     a->hit_count++;
00193   }
00194   
00195   return a->cache;
00196 }
00197 #endif /* HAVE_INLINE */
