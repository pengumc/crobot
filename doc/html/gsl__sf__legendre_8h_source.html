00105 /* P_l^m(x)  m >= 0; l >= m; |x| <= 1.0
00106  *
00107  * Note that this function grows combinatorially with l.
00108  * Therefore we can easily generate an overflow for l larger
00109  * than about 150.
00110  *
00111  * There is no trouble for small m, but when m and l are both large,
00112  * then there will be trouble. Rather than allow overflows, these
00113  * functions refuse to calculate when they can sense that l and m are
00114  * too big.
00115  *
00116  * If you really want to calculate a spherical harmonic, then DO NOT
00117  * use this. Instead use legendre_sphPlm() below, which  uses a similar
00118  * recursion, but with the normalized functions.
00119  *
00120  * exceptions: GSL_EDOM, GSL_EOVRFLW
00121  */
00122 int     gsl_sf_legendre_Plm_e(const int l, const int m, const double x, gsl_sf_result * result);
00123 double  gsl_sf_legendre_Plm(const int l, const int m, const double x);

00257 /* The following spherical functions are specializations
00258  * of Legendre functions which give the regular eigenfunctions
00259  * of the Laplacian on a 3-dimensional hyperbolic space.
00260  * Of particular interest is the flat limit, which is
00261  * Flat-Lim := {lambda->Inf, eta->0, lambda*eta fixed}.
00262  */
00263   
00264 /* Zeroth radial eigenfunction of the Laplacian on the
00265  * 3-dimensional hyperbolic space.
00266  *
00267  * legendre_H3d_0(lambda,eta) := sin(lambda*eta)/(lambda*sinh(eta))
00268  * 
00269  * Normalization:
00270  * Flat-Lim legendre_H3d_0(lambda,eta) = j_0(lambda*eta)
00271  *
00272  * eta >= 0.0
00273  * exceptions: GSL_EDOM
00274  */
00275 int gsl_sf_legendre_H3d_0_e(const double lambda, const double eta, gsl_sf_result * result);
00276 double gsl_sf_legendre_H3d_0(const double lambda, const double eta);
00277 
00278 
00279 /* First radial eigenfunction of the Laplacian on the
00280  * 3-dimensional hyperbolic space.
00281  *
00282  * legendre_H3d_1(lambda,eta) :=
00283  *    1/sqrt(lambda^2 + 1) sin(lam eta)/(lam sinh(eta))
00284  *    (coth(eta) - lambda cot(lambda*eta))
00285  * 
00286  * Normalization:
00287  * Flat-Lim legendre_H3d_1(lambda,eta) = j_1(lambda*eta)
00288  *
00289  * eta >= 0.0
00290  * exceptions: GSL_EDOM
00291  */
00292 int gsl_sf_legendre_H3d_1_e(const double lambda, const double eta, gsl_sf_result * result);
00293 double gsl_sf_legendre_H3d_1(const double lambda, const double eta);
